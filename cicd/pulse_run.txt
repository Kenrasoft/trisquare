pipeline {
    agent any
    
    stages {
        stage('Pull and Run Image') {
            steps {
                script {
                    git branch: 'main', url: 'https://github.com/kenrasoftorg/trisquare.git'
                    def imageTag = readFile('C:/Users/avina/.jenkins/workspace/pulse_build/pulseversion.txt').trim() // Read the image tag from version.txt
                    withCredentials([string(credentialsId: 'docker-hub-password', variable: 'DOCKER_PASSWORD')]) {
                        bat """
                            docker login -u nashkenra -p '$DOCKER_PASSWORD'
                            docker pull nashkenra/kenrasoft:${imageTag}
                            docker run -d --name my_pulse nashkenra/kenrasoft:${imageTag}
                        """
                    }
                }
            }
        }
    }
}
